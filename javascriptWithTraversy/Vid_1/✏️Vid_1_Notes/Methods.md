#### Length

```javascript

let s = "hello world"
console.log(s.length);
// result = 11
// space is counted

```

#### toUpperCase

```javascript

let s = "hello world"
console.log(s.toUpperCase);
// result = "HELLO WORLD"

```

```javascript

let s = "technology, computers, it, code"
console.log(s.split(","));

```
